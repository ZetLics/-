def demonstrate_dict_merge():
    """–î–µ–º–æ–Ω—Å—Ç—Ä–∞—Ü–∏—è –º–µ—Ç–æ–¥–æ–≤ –æ–±—ä–µ–¥–∏–Ω–µ–Ω–∏—è —Å–ª–æ–≤–∞—Ä–µ–π"""

    dict_1 = {1: 'a', 2: 'b'}
    dict_2 = {2: 'c', 4: 'd'}

    # –ú–µ—Ç–æ–¥ update()
    dict_1_copy = dict_1.copy()
    dict_1_copy.update(dict_2)
    print(f"–ú–µ—Ç–æ–¥ update():\n{dict_1_copy}")

    # –†–∞—Å–ø–∞–∫–æ–≤–∫–∞ —Å–ª–æ–≤–∞—Ä–µ–π (Python 3.9+)
    new_dict = {**dict_1, **dict_2}
    print(f"\n–†–∞—Å–ø–∞–∫–æ–≤–∫–∞ —Å–ª–æ–≤–∞—Ä–µ–π:\n{new_dict}")

# ================= 2. –û–ë–†–ê–ë–û–¢–ö–ê –°–õ–û–í–ê–†–ï–ô: –ú–ê–ö–°–ò–ú–£–ú –ò –°–£–ú–ú–ê ================

from collections import defaultdict

def max_dict(*dicts):
    """–û–±—ä–µ–¥–∏–Ω—è–µ—Ç —Å–ª–æ–≤–∞—Ä–∏, –≤—ã–±–∏—Ä–∞—è –º–∞–∫—Å–∏–º–∞–ª—å–Ω—ã–µ –∑–Ω–∞—á–µ–Ω–∏—è –¥–ª—è –∫–ª—é—á–µ–π"""
    merged = {}
    for d in dicts:
        for key, value in d.items():
            merged[key] = max(value, merged.get(key, value))
    return merged

def sum_dict(*dicts):
    """–°—É–º–º–∏—Ä—É–µ—Ç –∑–Ω–∞—á–µ–Ω–∏—è –¥–ª—è –ø–æ–≤—Ç–æ—Ä—è—é—â–∏—Ö—Å—è –∫–ª—é—á–µ–π"""
    summed = defaultdict(int)
    for d in dicts:
        for key, value in d.items():
            summed[key] += value
    return dict(summed)

def process_dictionaries():
    """–î–µ–º–æ–Ω—Å—Ç—Ä–∞—Ü–∏—è —Ä–∞–±–æ—Ç—ã —Å –æ–±—Ä–∞–±–æ—Ç–∫–æ–π —Å–ª–æ–≤–∞—Ä–µ–π"""
    print("\n=== 2. –û–ë–†–ê–ë–û–¢–ö–ê –°–õ–û–í–ê–†–ï–ô ===")
    
    dict_1 = {1: 12, 2: 33, 3: 10, 4: 10, 5: 2, 6: 90}
    dict_2 = {1: 12, 3: 7, 4: 1, 5: 2, 7: 112}
    dict_3 = {2: 3, 3: 3, 4: 60, 6: 8, 7: 25, 8: 71}
    dict_4 = {3: 1, 4: 13, 5: 31, 9: 9, 10: 556}

    print("–ú–∞–∫—Å–∏–º—É–º—ã (dict1 + dict2):", max_dict(dict_1, dict_2))
    print("–°—É–º–º—ã (dict1 + dict3 + dict4):", sum_dict(dict_1, dict_3, dict_4))
    print("–ü–æ–ª–Ω—ã–µ –º–∞–∫—Å–∏–º—É–º—ã:", max_dict(dict_1, dict_2, dict_3, dict_4))
    print("–ü–æ–ª–Ω—ã–µ —Å—É–º–º—ã:", sum_dict(dict_1, dict_2, dict_3, dict_4))

def set_gen(numbers):
    """–ì–µ–Ω–µ—Ä–∏—Ä—É–µ—Ç –º–Ω–æ–∂–µ—Å—Ç–≤–æ —Å –æ—Å–æ–±—ã–º–∏ –ø—Ä–∞–≤–∏–ª–∞–º–∏ –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏—è"""
    count = {}
    result = []
    for num in numbers:
        count[num] = count.get(num, 0) + 1
        result.append(num if count[num] == 1 else str(num) * count[num])
    return set(result)

def demonstrate_set_generation():
    """–î–µ–º–æ–Ω—Å—Ç—Ä–∞—Ü–∏—è –≥–µ–Ω–µ—Ä–∞—Ü–∏–∏ –º–Ω–æ–∂–µ—Å—Ç–≤"""

    lists = [

        [5, 5, 5, 5, 5, 5, 5],

    ]

def superset(a, b):
    """–ê–Ω–∞–ª–∏–∑–∏—Ä—É–µ—Ç –æ—Ç–Ω–æ—à–µ–Ω–∏–µ –º–Ω–æ–∂–µ—Å—Ç–≤ –∏ –≤—ã–≤–æ–¥–∏—Ç —Ä–µ–∑—É–ª—å—Ç–∞—Ç"""
    if a == b:
        print(f"–ú–Ω–æ–∂–µ—Å—Ç–≤–∞ —Ä–∞–≤–Ω—ã: {a} == {b}")
    elif a > b:
        print(f"–û–±—ä–µ–∫—Ç {a} —è–≤–ª—è–µ—Ç—Å—è —á–∏—Å—Ç—ã–º —Å—É–ø–µ—Ä–º–Ω–æ–∂–µ—Å—Ç–≤–æ–º –¥–ª—è {b}")
    elif b > a:
        print(f"–û–±—ä–µ–∫—Ç {b} —è–≤–ª—è–µ—Ç—Å—è —á–∏—Å—Ç—ã–º —Å—É–ø–µ—Ä–º–Ω–æ–∂–µ—Å—Ç–≤–æ–º –¥–ª—è {a}")
    else:
        print(f"–°—É–ø–µ—Ä–º–Ω–æ–∂–µ—Å—Ç–≤–æ –Ω–µ –æ–±–Ω–∞—Ä—É–∂–µ–Ω–æ: {a} vs {b}")

def demonstrate_superset():
    """–î–µ–º–æ–Ω—Å—Ç—Ä–∞—Ü–∏—è –∞–Ω–∞–ª–∏–∑–∞ –º–Ω–æ–∂–µ—Å—Ç–≤"""

        ({1, 8, 3, 5}, {3, 5}),
        ({1, 8, 3, 5}, {5, 3, 8, 1}),
        ({3, 5}, {5, 3, 8, 1}),
        ({90, 100}, {3, 5})
    ]
    
    for a, b in sets:
        superset(a, b)
